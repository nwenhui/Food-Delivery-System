TABLE Customers {
 Cid INTEGER PK
	name VARCHAR // NOT NULL
	numOrders INTEGER
	rewardPoints INTEGER
	currentAddress VARCHAR // NOT NULL
	lastOrderTime DATETIME
	createdDate DATE // NOT NULL
	creditCardNumber VARCHAR
	accessRight INTEGER // NOT NULL
}

TABLE Restaurants {
	Rid INTEGER PK
	name VARCHAR(50) // NOT NULL
	minAmount DECIMAL
	address VARCHAR(100) // NOT NULL
}

TABLE Staffs {
 Sid INTEGER PK
 Rid INTEGER // NOT NULL
	name VARCHAR(50) // NOT NULL
	hireDate DATE // NOT NULL
	terminationDate DATE
	accessRight INTEGER // NOT NULL
}

Ref: Staffs.Sid > Restaurants.Rid

TABLE FoodItem {
	Fid INTEGER PK
	Rid INTEGER // NOT NULL
	name VARCHAR(50) // NOT NULL
	originalPrice DECIMAL // NOT NULL
	categories VARCHAR(10) // NOT NULL
	dailyLimit INTEGER // NOT NULL
	availabilityStatus INTEGER // to reset to dailyLimit daily
}

Ref: FoodItem.Fid > Restaurants.Rid


TABLE DeliveryRiders {
	Did INTEGER PK
	name VARCHAR(50)
	startDate DATE
	terminationDate DATE
	baseSalary DECIMAL //Null delivery fee to be added based on the number of deliveries made in that month
	employmentType INTEGER // Not null 1: fullTime, 2: partTime
	accessRight INTEGER //NOT NULL shifts?
}

TABLE Orders {
	Oid INTEGER PK
	Did INTEGER //NOT NULL 
	Cid INTEGER// NOT NULL,
	cost DECIMAL// NOT NULL
	location VARCHAR(100) // NOT NULL,
	orderTime TIMESTAMP// NOT NULL,
	deliveryCost DECIMAL// NOT NULL,
}

Ref: Orders.Did - DeliveryRiders.Did
Ref: Orders.Cid - Customers.Cid
